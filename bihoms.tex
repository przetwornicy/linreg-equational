\subsection{Completeness}

Following the notation from the previous sections, we write $f \equiv g$ if the equality of $f$ and $g$ can be proved using rules from \cref{subsec:simple_functions_rules}.

\begin{lemma}\label{lem:simple_functions_completeness}
    For every pair of simple functions $f,g : A \to \Gamma^*$, if $f = g$, then $f \equiv g$. 
\end{lemma}

The rest of this section is devoted to proving \cref{lem:simple_functions_completeness}. We proceed by a nested induction. The outer induction is on the rank of type $A$ \katzpermargin{define the rank of type}. Once the type is fixed, we write
\begin{gather*}
    f = \delta_0 \cdot f_1 \cdot \delta_1 \cdots \delta_{n-1} \cdot f_n \cdot \delta_n \\ 
    g = \gamma_0 \cdot g_1 \cdot \gamma_1 \cdots \gamma_{m-1} \cdot g_m \cdot \gamma_m,
\end{gather*}
where $f_1, \dots, f_n, g_1, \dots, g_m$ are bihomomorphisms ($\leftarrow$ or $\rightarrow$) on one of the root lists, and $\delta_0, \dots, \delta_n, \gamma_0, \dots, \gamma_m$ are functions on $(\first{A}, \last{A})$. Then, the inner induction is on $n+m$. 

We assume that no $f_i$, $g_j$ is equal to $\varepsilon$, otherwise we can eliminate it.

We start with some simple cases. If $A = 1$, then the only possible functions are constant, so we actually need a rule for them? \katzpermargin{how to resolve constant functions?} If $n = 0$ or $m = 0$, then it must hold that either $n = m = 0$, in which case we reduce to a simpler type, or there is some $f_i = \varepsilon$ (or $g_j = \varepsilon$), which we can eliminate.

First, we show that only three cases can occur: (1) $n = m = 1$, (2) all $f_1, \dots, f_n,g_1, \dots, g_m$ are unary, or (3) we can find a proper good interface and split reasoning into two separate parts. 

By $f[i:j]$ we denote the function $f_i \cdot \delta_i \cdots \delta_{j-1} \cdot f_j$, and analogously we define $g[i:j]$. If $i > j$, then $f[i:j] = \varepsilon$.

\begin{lemma}\label{lemma:the-only-three-cases-for-simple-functions}
    Suppose that $f = g$ and $n,m \geq 1$. Then at least one of the following must hold:
    \begin{enumerate}
        \item $n = m = 1$,
        \item there exists $u \in \Gamma^*$ such that $\im(f[1:n]), \im(g[1:m]) \subseteq_{\mathrm{infix}} u^\omega$, or
        \item there exist $i \in [n], j \in [m]$, such that $1 \leq i + j < n+m$ and the right-interface of $\delta_0 \cdot f[1: i]$ and $\gamma_0 \cdot g[1:j]$ is good.
    \end{enumerate}
\end{lemma}

\subsection{A proof of \cref{lemma:the-only-three-cases-for-simple-functions}}

First, we formally define what we mean that a simple function is unary.

\begin{definition}[Unary simple functions]
    We say that $f : A \xrightarrow{\mathrm{sf}} \Gamma^*$ is $u$-unary, for some prime $u \in \Gamma^*$, if there exist mappings $p,s : \shell{A} \to \{0, \dots, |u|-1 \}$ (they depend only on the shell!), such that
    \[
        \substr{u}{}{p} \cdot f \cdot \substr{u}{|u| - s + 1}{} \in u^*.
    \]
\end{definition}

\begin{claim}
    Suppose that $f : A \xrightarrow{\mathrm{sf}} \Gamma^*$ is a bihomomorphism, $f \neq \varepsilon$, $u \in \Gamma^*$ is prime, and $p : \shell{A} \to \{0, \dots, |u|-1\}$ is such that $\substr{u}{}{p} \cdot f \leq_\prefix u^\omega$. Then $f$ is $u$-unary.
\end{claim}

\begin{proof}
    We can assume that $A$ is a list type. Let $\overline x : A$. It suffices to show that $|f(\overline x)| \, (\mathrm{mod} \, |u|)$ depends only on $\shell{\overline x}$. Assume that $f = \bihom{to, \cdot}$ (the case $\leftarrow$ is symmetric). Take $\overline y : A$ such that $\first{\overline y} = \last{\overline x}$, and $\last{\overline y} = \first{\overline x}$, and $|f(\overline y)| \geq 3 \cdot |u|$. Consider the string $S = \substr{u}{}{p(\shell{\overline x})} \cdot f(\overline x \cdot \overline y) = \substr{u}{}{p(\shell{\overline x})} \cdot f(\overline x)  \cdot f(\overline y)$. We know that:
    \begin{itemize}
        \item $S \leq_\prefix u^\omega$,
        \item For $\Delta = |u| - p(\shell{\overline y})$, we have $\substr{f(\overline y)}{\Delta + 1}{} \leq_\prefix u^\omega$.
    \end{itemize}

    The second bullet gives $\substr{S}{p(\shell{\overline x}) + |f(\overline x)| + \Delta + 1}{} \leq_\prefix u^\omega$. Combining this with the first bullet, we deduce that if $p + |f| + \Delta$ is not divisible by $|u|$, then we have $u \leq_\infix u^2$, and $u$ is not prime by \cref{lem:prime-words-must-perfectly-align} (see \cref{fig:f-modulo-u-depends-only-on-shell}).

    \begin{figure}[h]
        \centering
        \includegraphics[width=1\linewidth]{pictures/unary_bihoms_prefix_gives_suffix.pdf}
        \caption{If $p + |f| + \Delta$ is not divisible by $|u|$, then we have misalignment, yielding that $u$ is not prime.}
        \label{fig:f-modulo-u-depends-only-on-shell}
    \end{figure}

    Since $\shell{\overline y}$ (and hence $\Delta$) depends only on $\shell{\overline x}$, we deduce that $|f(\overline x)| \, (\mathrm{mod} \, |u|)$ also depends only on $\shell{\overline x}$, which finishes the proof.
\end{proof}

\begin{claim}
    Suppose that $f_1, f_2 : A \xrightarrow{sf} \Gamma^*$ are both $u$-unary, and their mappings are $p_i, s_i$ for $i \in \{1,2\}$. If
    \[
        p_2 + s_1 = |u|. 
    \]
    then $f_1 \cdot f_2$ is $u$-unary.
\end{claim}

\begin{proof}
    We show that the mappings $p := p_1, s := s_2$ work. Let $\overline x : A$ be arbitrary. Since $\substr{u}{}{p_1} \cdot f_1 \leq_\prefix u^\omega$, we have $\substr{u}{}{p_1(\overline x)} \cdot f_1(\overline x) = u^k \cdot \substr{u}{}{|u| - s_1(\overline x)}$ for some $k \geq 0$. Similarly, $f_2(\overline x) \cdot \substr{u}{|u|- s_2(\overline x) + 1}{} = \substr{u}{p_2(\overline x)+1}{} \cdot u^\ell$ for some $\ell \geq 0$. Thus,
    \begin{align*}
             &\left( \substr{u}{}{p} \cdot f_1 \right) \cdot \left(f_2 \cdot \substr{u}{|u|- s + 1}{}\right) \\
        = \, &u^k \cdot \substr{u}{}{|u| - s_1} \cdot \substr{u}{p_2 + 1}{} \cdot u^\ell \\
        = \, &u^k \cdot \substr{u}{}{|u| - s_1} \cdot \substr{u}{|u| - s_1 + 1}{} \cdot u^\ell \\ 
        \in \, &u^* \qedhere
    \end{align*}
\end{proof}
